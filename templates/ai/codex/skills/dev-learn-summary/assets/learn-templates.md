## Script template

```bash
#!/usr/bin/env bash
set -euo pipefail

usage() {
  cat <<'USAGE'
Usage: bin/scripts/<script-name>.sh <REQUIRED_1> <REQUIRED_2>

Optional environment overrides:
  OPTIONAL_1  Default: <value>
USAGE
}

if [[ "${1:-}" == "-h" || "${1:-}" == "--help" ]]; then
  usage
  exit 0
fi

if [[ $# -ne 2 ]]; then
  usage
  exit 1
fi

REQUIRED_1="$1"
REQUIRED_2="$2"
OPTIONAL_1="${OPTIONAL_1:-<value>}"

if ! command -v gcloud >/dev/null 2>&1; then
  echo "gcloud not found." >&2
  exit 1
fi

# Do work
```
